    <script>

        const SUPABASE_URL = 'https://apwgdjzwhnwjdbonaway.supabase.co';

        const SUPABASE_KEY = 'eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9.eyJpc3MiOiJzdXBhYmFzZSIsInJlZiI6ImFwd2dkanp3aG53amRib25hd2F5Iiwicm9sZSI6ImFub24iLCJpYXQiOjE3NzU0MzY2MjcsImV4cCI6MjA5MTAxMjYyN30._ZBC_NKIaUNQgzE2A4U3uw7cLy2_O6eshif7LzcIfLg';

        const supabaseClient = supabase.createClient(SUPABASE_URL, SUPABASE_KEY);


        const form = document.getElementById('expense-form');

        const list = document.getElementById('expense-list');

        const totalDisplay = document.getElementById('total-amount');


        async function fetchExpenses() {

            const { data, error } = await supabaseClient

                .from('expenses')

                .select('*')

                .order('created_at', { ascending: false });


            if (error) {

                console.error('Error fetching:', error);

                return;

            }

            renderExpenses(data);

        }


        function renderExpenses(expenses) {

            list.innerHTML = '';

            let total = 0;

            expenses.forEach(item => {

                total += parseFloat(item.amount);

                const div = document.createElement('div');

                div.className = 'glass-card p-4 rounded-xl flex justify-between items-center shadow-sm';

                div.innerHTML = `

                    <div>

                        <p class="font-bold text-gray-800">${item.category}</p>

                        <p class="text-xs text-gray-500">${item.description || 'No description'} • ${item.spender}</p>

                    </div>

                    <div class="text-right">

                        <p class="font-bold text-blue-600">HK$ ${parseFloat(item.amount).toFixed(2)}</p>

                        <p class="text-[10px] text-gray-400">${new Date(item.created_at).toLocaleDateString()}</p>

                    </div>`;

                list.appendChild(div);

            });

            totalDisplay.innerText = `HK$ ${total.toFixed(2)}`;

        }


        form.addEventListener('submit', async (e) => {

            e.preventDefault();

            const newExpense = {

                amount: document.getElementById('amount').value,

                category: document.getElementById('category').value,

                spender: document.getElementById('spender').value,

                description: document.getElementById('description').value,

            };


            const { error } = await supabaseClient.from('expenses').insert([newExpense]);


            if (error) {

                alert('Error saving expense: ' + error.message);

            } else {

                form.reset();

                fetchExpenses();

            }

        });


        fetchExpenses();
